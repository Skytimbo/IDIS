        function createSparkle(x, y) {

            sparkle.style.top = y + 'px';
            sparkle.style.width = '4px';

            sparkle.style.borderRadius = '50%';

            sparkle.style.animation = 'sparkleAnimation 1s ease-out forwards';
            
            document.body.appendChild(sparkle);
            
            setTimeout(() => {
                sparkle.remove();
            }, 1000);
        }
